const int items_size = 3;
const int items[items_size] = {1, 2, 3};

for(int i=0; i<items_size; ++i)
  cout << items[i] << endl;

for(int i:items)
  cout << i << endl;

int i=0;

while(true)
{
  cout << items[i] << endl;

  if(++i == items_size)
    break;
}

i=0;
do {
  cout << items[i] << endl;
  ++i;
} while(i<items_size);

final int items[] = {1, 2, 3};

for(int i=0; i<items.length; ++i)
  System.out.println(i);

for(int i:items)
  System.out.println(i);

int i=0;

while(true) {
  System.out.println(i);
  ++i;

  if(i==items.length)
    break;
}

i=0;
do {
  System.out.println(i);
  ++i;
} while(i<items.length);
